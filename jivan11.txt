Q-3 Write a program to print the first 10 natural numbers in reverse order.
Ans =
     #include<stdio.h>
     
#include<conio.h>

     
int main()
     
{
         int i=10;
     
    while(i>=1)

         {

             printf("\n%d",i);
         
    i--;
     
    }

         getch();
      
}
